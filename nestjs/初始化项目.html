# 初始化项目

https://docs.nestjs.comopen in new window

https://docs.nestjs.cnopen in new window

# install nest cli
npm i -g @nestjs/cli

# new project
nest new project_name

# development mode
npm rum start:dev

# Add TypeORM
npm install --save @nestjs/typeorm typeorm mysql2

# generate module
nest g resource modules/xxx
# 不生成测试文件
nest g resource modules/xxx --no-spec

# 创建 modules 目录
mkdir src/modules
